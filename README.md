        function updateClock() {
            // Create a date object for current time in Dhaka
            const options = {
                timeZone: 'Asia/Dhaka',
                hour12: false,
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            };
            
            const dateOptions = {
                timeZone: 'Asia/Dhaka',
                weekday: 'long',
                year: 'numeric',
                month: 'long',
                day: 'numeric'
            };
            
            const now = new Date();
            
            // Format time (HH:MM:SS)
            let timeStr = now.toLocaleTimeString('en-US', options);
            // Add blinking effect to time separators
            timeStr = timeStr.replace(/:/g, '<span class="time-separator">:</span>');
            
            // Format date (Weekday, Month Day, Year)
            const dateStr = now.toLocaleDateString('en-US', dateOptions);
            
            // Update the DOM
            document.getElementById('time').innerHTML = timeStr;
            document.getElementById('date').textContent = dateStr;
        }
        
        // Update the clock immediately and then every second
        updateClock();
        setInterval(updateClock, 1000);
    </script>

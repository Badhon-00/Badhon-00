<script>
  // Function to update Bangladesh time
  function updateBangladeshTime() {
    const now = new Date();
    
    // Convert to Bangladesh Time (UTC+6)
    const bangladeshTime = new Date(now.getTime() + (6 * 60 * 60 * 1000));
    
    // Format time
    const timeOptions = { 
      timeZone: 'Asia/Dhaka',
      hour12: true,
      hour: 'numeric',
      minute: 'numeric',
      second: 'numeric'
    };
    
    // Format date
    const dateOptions = { 
      timeZone: 'Asia/Dhaka',
      weekday: 'long',
      year: 'numeric',
      month: 'long',
      day: 'numeric'
    };
    
    const timeStr = bangladeshTime.toLocaleTimeString('en-US', timeOptions);
    const dateStr = bangladeshTime.toLocaleDateString('en-US', dateOptions);
    
    // Update elements
    document.getElementById('time').textContent = timeStr;
    document.getElementById('date').textContent = dateStr;
  }
  
  // Update last updated time
  function updateLastUpdated() {
    const now = new Date();
    const options = { 
      year: 'numeric', 
      month: 'long', 
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit'
    };
    document.getElementById('last-updated').textContent = now.toLocaleDateString('en-US', options);
  }
  
  // Initialize
  updateBangladeshTime();
  updateLastUpdated();
  
  // Update time every second
  setInterval(updateBangladeshTime, 1000);
</script>
